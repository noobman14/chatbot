## 一、数组 ⭐⭐⭐

### 1. 两数之和 ⭐⭐⭐

**题目**: 给定数组和目标值，找出和为目标值的两个数的下标

```javascript
// 暴力解法 O(n²)
function twoSum(nums, target) {
  for (let i = 0; i < nums.length; i++) {
    for (let j = i + 1; j < nums.length; j++) {
      if (nums[i] + nums[j] === target) {
        return [i, j];
      }
    }
  }
  return [];
}

// 哈希表解法 O(n) ✅ 推荐
function twoSum(nums, target) {
  const map = new Map();
  
  for (let i = 0; i < nums.length; i++) {
    const complement = target - nums[i];
    
    if (map.has(complement)) {
      return [map.get(complement), i];
    }
    
    map.set(nums[i], i);
  }
  
  return [];
}

// 测试
console.log(twoSum([2, 7, 11, 15], 9)); // [0, 1]
```

**思路**: 用 Map 存储已遍历的数，每次检查 `target - 当前数` 是否存在

### 2. 三数之和 ⭐⭐

**题目**: 找出所有和为 0 的三元组（不重复）

```javascript
function threeSum(nums) {
  const result = [];
  nums.sort((a, b) => a - b);  // 先排序
  
  for (let i = 0; i < nums.length - 2; i++) {
    // 跳过重复的第一个数
    if (i > 0 && nums[i] === nums[i - 1]) continue;
    
    // 双指针
    let left = i + 1;
    let right = nums.length - 1;
    
    while (left < right) {
      const sum = nums[i] + nums[left] + nums[right];
      
      if (sum === 0) {
        result.push([nums[i], nums[left], nums[right]]);
        
        // 跳过重复
        while (left < right && nums[left] === nums[left + 1]) left++;
        while (left < right && nums[right] === nums[right - 1]) right--;
        
        left++;
        right--;
      } else if (sum < 0) {
        left++;
      } else {
        right--;
      }
    }
  }
  
  return result;
}

// 测试
console.log(threeSum([-1, 0, 1, 2, -1, -4])); // [[-1, -1, 2], [-1, 0, 1]]
```

**思路**: 排序 + 固定一个数 + 双指针

### 3. 合并两个有序数组 ⭐⭐⭐

```javascript
function merge(nums1, m, nums2, n) {
  let i = m - 1;      // nums1 的末尾
  let j = n - 1;      // nums2 的末尾
  let k = m + n - 1;  // 合并后的末尾
  
  // 从后往前填充
  while (j >= 0) {
    if (i >= 0 && nums1[i] > nums2[j]) {
      nums1[k--] = nums1[i--];
    } else {
      nums1[k--] = nums2[j--];
    }
  }
  
  return nums1;
}

// 测试
console.log(merge([1, 2, 3, 0, 0, 0], 3, [2, 5, 6], 3)); 
// [1, 2, 2, 3, 5, 6]
```

**思路**: 从后往前填充，避免覆盖

```javascript
function moveZeroes(nums) {
  let insertPos = 0;
  
  // 把非零元素移到前面
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] !== 0) {
      nums[insertPos++] = nums[i];
    }
  }
  
  // 后面填充零
  while (insertPos < nums.length) {
    nums[insertPos++] = 0;
  }
  
  return nums;
}
```

```javascript
// 方法1: Set
function findDuplicate(nums) {
  const set = new Set();
  for (const num of nums) {
    if (set.has(num)) return num;
    set.add(num);
  }
  return -1;
}

// 方法2: 原地标记（O(1)空间）
function findDuplicate2(nums) {
  for (let i = 0; i < nums.length; i++) {
    const index = Math.abs(nums[i]);
    if (nums[index] < 0) {
      return index;
    }
    nums[index] = -nums[index];
  }
  return -1;
}
```

```javascript
function isValid(s) {
  const stack = [];
  const map = {
    ')': '(',
    ']': '[',
    '}': '{'
  };
  
  for (const char of s) {
    if (char in map) {
      // 右括号：检查栈顶是否匹配
      if (stack.pop() !== map[char]) {
        return false;
      }
    } else {
      // 左括号：入栈
      stack.push(char);
    }
  }
  
  return stack.length === 0;
}

// 测试
```

### 3. 最长回文子串 ⭐⭐

```javascript
function longestPalindrome(s) {
  if (s.length < 2) return s;
  
  let start = 0;
  let maxLen = 1;
  
  // 中心扩展
  function expandAroundCenter(left, right) {
    while (left >= 0 && right < s.length && s[left] === s[right]) {
      const len = right - left + 1;
      if (len > maxLen) {
        maxLen = len;
        start = left;
      }
      left--;
      right++;
    }
  }
  
  for (let i = 0; i < s.length; i++) {
    expandAroundCenter(i, i);      // 奇数长度
    expandAroundCenter(i, i + 1);  // 偶数长度
  }
  
  return s.substring(start, start + maxLen);
}

// 测试
console.log(longestPalindrome('babad')); // 'bab' 或 'aba'
```

**思路**: 从每个位置向两边扩展，找最长回文

### 4. 无重复字符的最长子串 ⭐⭐⭐

```javascript
function lengthOfLongestSubstring(s) {
  const charIndex = new Map();
  let maxLen = 0;
  let start = 0;
  
  for (let end = 0; end < s.length; end++) {
    const char = s[end];
    
    // 如果字符重复了，移动起点
    if (charIndex.has(char) && charIndex.get(char) >= start) {
      start = charIndex.get(char) + 1;
    }
    
    charIndex.set(char, end);
    maxLen = Math.max(maxLen, end - start + 1);
  }
  
  return maxLen;
}

// 测试
console.log(lengthOfLongestSubstring('abcabcbb')); // 3 ('abc')
console.log(lengthOfLongestSubstring('pwwkew'));   // 3 ('wke')
```

**思路**: 滑动窗口 + Map记录字符位置

### 5. 字符串转整数 (atoi) ⭐⭐

```javascript
function myAtoi(s) {
  const INT_MAX = 2 ** 31 - 1;
  const INT_MIN = -(2 ** 31);
  
  let i = 0;
  let sign = 1;
  let result = 0;
  
  // 1. 跳过前导空格
  while (s[i] === ' ') i++;
  
  // 2. 处理符号
  if (s[i] === '+' || s[i] === '-') {
    sign = s[i] === '-' ? -1 : 1;
    i++;
  }
  
  // 3. 读取数字
  while (i < s.length && /\d/.test(s[i])) {
    result = result * 10 + parseInt(s[i]);
    i++;
    
    // 提前判断溢出
    if (result * sign > INT_MAX) return INT_MAX;
    if (result * sign < INT_MIN) return INT_MIN;
  }
  
  return result * sign;
}

// 测试
console.log(myAtoi('42'));           // 42
console.log(myAtoi('   -42'));       // -42
console.log(myAtoi('4193 with'));    // 4193
```

```javascript
// 迭代法
function reverseList(head) {
  let prev = null;
  let curr = head;
  
  while (curr) {
    const next = curr.next;  // 保存下一个
    curr.next = prev;        // 反转指向
    prev = curr;             // prev 前进
    curr = next;             // curr 前进
  }
  
  return prev;
}

// 递归法
function reverseListRecursive(head) {
  if (!head || !head.next) return head;
  
  const newHead = reverseListRecursive(head.next);
  head.next.next = head;
  head.next = null;
  
  return newHead;
}
```

```javascript
function mergeTwoLists(l1, l2) {

    if (l1.val <= l2.val) {
      curr.next = l1;
```

```javascript
// 判断是否有环
function hasCycle(head) {
  let slow = head;
  let fast = head;
  
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    
    if (slow === fast) {
      return true;
    }
  }
  
  return false;
}

// 找环的入口
function detectCycle(head) {
  let slow = head;
  let fast = head;
  
  // 先找相遇点
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    
    if (slow === fast) {
      // 找入口：一个从头开始，一个从相遇点开始
      let ptr = head;
      while (ptr !== slow) {
        ptr = ptr.next;
        slow = slow.next;
      }
      return ptr;
    }
  }
  
  return null;
}
```

**思路**: 快慢指针（龟兔赛跑）

### 4. 相交链表 ⭐⭐

```javascript
function getIntersectionNode(headA, headB) {
  if (!headA || !headB) return null;
  
  let pA = headA;
  let pB = headB;
  
  // 两个指针走完自己的再走对方的，最终会在交点相遇
  while (pA !== pB) {
    pA = pA ? pA.next : headB;
    pB = pB ? pB.next : headA;
  }
  
  return pA;
}
```

**思路**: 两指针先走自己的链表，到末尾后走另一个，总路程相等时会在交点相遇

## 四、栈/队列 ⭐⭐

### 1. 用栈实现队列 ⭐⭐

```javascript
class MyQueue {
  constructor() {
    this.inStack = [];   // 入队栈
    this.outStack = [];  // 出队栈
  }
  
  push(x) {
    this.inStack.push(x);
  }
  
  pop() {
    if (this.outStack.length === 0) {
      // 把入队栈的元素倒入出队栈
      while (this.inStack.length) {
        this.outStack.push(this.inStack.pop());
      }
    }
    return this.outStack.pop();
  }
  
  peek() {
    if (this.outStack.length === 0) {
      while (this.inStack.length) {
        this.outStack.push(this.inStack.pop());
      }
    }
    return this.outStack[this.outStack.length - 1];
  }
  
  empty() {
    return this.inStack.length === 0 && this.outStack.length === 0;
  }
}
```

### 2. 最小栈 ⭐⭐

```javascript
class MinStack {
  constructor() {
    this.stack = [];
    this.minStack = [];  // 辅助栈，存储最小值
  }
  
  push(val) {
    this.stack.push(val);
    // 如果辅助栈为空或当前值更小，入辅助栈
    if (this.minStack.length === 0 || val <= this.getMin()) {
      this.minStack.push(val);
    }
  }
  
  pop() {
    const val = this.stack.pop();
    // 如果弹出的是最小值，辅助栈也弹出
    if (val === this.getMin()) {
      this.minStack.pop();
    }
    return val;
  }
  
  top() {
    return this.stack[this.stack.length - 1];
  }
  
  getMin() {
    return this.minStack[this.minStack.length - 1];
  }
}
```

### 3. 每日温度 ⭐⭐

**题目**: 找出比当天温度高的最近一天是几天后

```javascript
function dailyTemperatures(temperatures) {
  const n = temperatures.length;
  const result = new Array(n).fill(0);
  const stack = [];  // 存索引
  
  for (let i = 0; i < n; i++) {
    // 当前温度比栈顶高，计算栈顶等待天数
    while (stack.length && temperatures[i] > temperatures[stack[stack.length - 1]]) {
      const prevIndex = stack.pop();
      result[prevIndex] = i - prevIndex;
    }
    stack.push(i);
  }
  
  return result;
}

// 测试
```

### 1. 二叉树的遍历 ⭐⭐⭐

```javascript
// 前序遍历：根 -> 左 -> 右
function preorder(root) {
  if (!root) return [];
  return [root.val, ...preorder(root.left), ...preorder(root.right)];
}

// 中序遍历：左 -> 根 -> 右
function inorder(root) {
  if (!root) return [];
  return [...inorder(root.left), root.val, ...inorder(root.right)];
}

// 后序遍历：左 -> 右 -> 根
function postorder(root) {
  if (!root) return [];
  return [...postorder(root.left), ...postorder(root.right), root.val];
}

// 层序遍历 (BFS)
function levelOrder(root) {
  if (!root) return [];
  
  const result = [];
  const queue = [root];
  
  while (queue.length) {
    const levelSize = queue.length;
    const level = [];
    
    for (let i = 0; i < levelSize; i++) {
      const node = queue.shift();
      level.push(node.val);
      
      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }
    
    result.push(level);
  }
  
  return result;
}
```

```javascript
function isSymmetric(root) {
  if (!root) return true;
  
  function isMirror(left, right) {
    if (!left && !right) return true;
    if (!left || !right) return false;
    
    return left.val === right.val 
      && isMirror(left.left, right.right) 
      && isMirror(left.right, right.left);
  }
  
  return isMirror(root.left, root.right);
}
```

### 4. 二叉树的最近公共祖先 ⭐⭐

```javascript
function lowestCommonAncestor(root, p, q) {
  if (!root || root === p || root === q) return root;
  
  const left = lowestCommonAncestor(root.left, p, q);
  const right = lowestCommonAncestor(root.right, p, q);
  
  // 如果左右都找到，当前节点就是祖先
  if (left && right) return root;
  
  // 否则返回找到的那个
  return left || right;
}
```

```javascript
function climbStairs(n) {
  if (n <= 2) return n;
  
  let prev1 = 1;  // f(n-2)
  let prev2 = 2;  // f(n-1)
  
  for (let i = 3; i <= n; i++) {
    const curr = prev1 + prev2;
    prev1 = prev2;

}
```

### 3. 零钱兑换 ⭐⭐

```javascript
function coinChange(coins, amount) {
  // dp[i] = 凑出金额i需要的最少硬币数
  const dp = new Array(amount + 1).fill(Infinity);
  dp[0] = 0;
  
  for (let i = 1; i <= amount; i++) {
    for (const coin of coins) {
      if (coin <= i && dp[i - coin] !== Infinity) {
        dp[i] = Math.min(dp[i], dp[i - coin] + 1);
      }
    }
  }
  
  return dp[amount] === Infinity ? -1 : dp[amount];
}
```

## 七、二分查找 ⭐⭐

### 基础模板

```javascript
function binarySearch(arr, target) {
  let left = 0;
  let right = arr.length - 1;
  
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    
    if (arr[mid] === target) {
      return mid;
    } else if (arr[mid] < target) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }
  
  return -1;  // 未找到
}
```

```javascript
function searchLeft(arr, target) {
```
